# Steam launch options for NVIDIA + DLL mods

# Basic NVIDIA/NVAPI options:
PROTON_ENABLE_NVAPI=1 PROTON_HIDE_NVIDIA_GPU=0 DXVK_ENABLE_NVAPI=1 %command%

# Common DLL mod options:
WINEDLLOVERRIDES="dxgi=n,b;dinput8=n,b;winmm=n,b" %command%

# NVIDIA + common DLL mods together:
PROTON_ENABLE_NVAPI=1 PROTON_HIDE_NVIDIA_GPU=0 DXVK_ENABLE_NVAPI=1 WINEDLLOVERRIDES="dxgi=n,b;dinput8=n,b;winmm=n,b" %command%

# For DX12/DXR games, sometimes useful:
PROTON_ENABLE_NVAPI=1 PROTON_HIDE_NVIDIA_GPU=0 DXVK_ENABLE_NVAPI=1 VKD3D_CONFIG=dxr %command%

# For DLSS/OptiScaler/DLSS Enabler mods, most common:
PROTON_ENABLE_NVAPI=1 PROTON_HIDE_NVIDIA_GPU=0 DXVK_ENABLE_NVAPI=1 WINEDLLOVERRIDES="dxgi=n,b" %command%
